Варіант 7
Метод дихотомії
Нелінійне рівняння:
\[2x^3 + 9x^2 - 10 = 0\]

Код написаний на C#

Program.cs


static double Function(double x)
{
    return 2 * x * x * x + 9 * x * x - 10;
}

static double Dichotomy(double a, double b, double tol)
{
    double c;

    while ((b - a) / 2.0 > tol)
    {
        c = (a + b) / 2.0;
        if (Function(c) == 0.0)
        {
            return c;
        }
        else if (Function(a) * Function(c) < 0)
        {
            b = c;
        }
        else
        {
            a = c;
        }
    }

    return (a + b) / 2.0;
}

double a = double.Parse(Console.ReadLine());
double b = double.Parse(Console.ReadLine());
double tol = 1e-6;
double root = Dichotomy(a, b, tol);

Console.WriteLine($"Корiнь рiвняння: {root:F7}");

Код написаний в Scilab

2_2.sce


function y = f(x)
    y = 2*x^3 + 9*x^2 - 10;
endfunction

function root = dichotomy(a, b, tol)
    if f(a) * f(b) >= 0 then
        error("Невірний початковий інтервал. Функція повинна мати різні знаки на кінцях інтервалу.");
    end

    while (b - a) / 2 > tol do
        c = (a + b) / 2;
        fc = f(c);

        if fc == 0 then
            root = c;
            return;
        elseif f(a) * fc < 0 then
            b = c;
        else
            a = c;
        end
    end

    root = (a + b) / 2;
endfunction

a = input("Введіть початок інтервалу (a): ");
b = input("Введіть кінець інтервалу (b): ");
tol = 1e-6;

if a >= b then
    error("Початок інтервалу повинен бути меншим за кінець інтервалу.");
end

root = dichotomy(a, b, tol);

disp("Корінь рівняння: " + string(root));
